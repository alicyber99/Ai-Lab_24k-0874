import random

states = ['Sunny', 'Cloudy', 'Rainy']

transition = {
    'Sunny': [0.6, 0.3, 0.1],
    'Cloudy': [0.3, 0.4, 0.3],
    'Rainy': [0.2, 0.3, 0.5]
}

current = 'Sunny'
days = []

for _ in range(10):
    days.append(current)
    current = random.choices(states, transition[current])[0]

print(days)
